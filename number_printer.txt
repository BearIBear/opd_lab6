; 0-9 = 0x0-0x9
; - 0xA
; + 0xB
; /  0xC
; * 0xD
; = 0xF

ORG 0
LD #0xFF
READ_NUMBER:
	IN 0x1D
	AND #0x40
	BEQ READ_NUMBER
IN 0x1C
PUSH
CALL $PRINT
POP
JUMP $READ_NUMBER
HLT

ORG 200

PRINT:
	LD &1
	CMP #0xA
	BPL SKIP_NUMBER
	PRINT_NUMBER:
		ADD #0x30
		OUT 0xC
		RET
	SKIP_NUMBER: BNE SKIP_MINUS
	PRINT_MINUS:
		LD #0x2D
		OUT 0xC
		RET
	SKIP_MINUS: CMP #0xB
	BNE SKIP_PLUS
	PRINT_PLUS:
		LD #0x2B
		OUT 0xC
		RET
	SKIP_PLUS: CMP #0xC
	BNE SKIP_DIVISION
	PRINT_DIVISION:
		LD #0x2F
		OUT 0xC
		RET
	SKIP_DIVISION: CMP #0xD
	BNE SKIP_MULTIPLICATION
	PRINT_MULTIPLICATION:
		LD #0x2A
		OUT 0xC
		RET
	SKIP_MULTIPLICATION: LD #0x3D
	OUT 0xC
	RET
